JS validacija

$(document).ready(function() {
    $("#taster").click(function() {
        let ime = document.querySelector("#ime").value;
        let mail = document.querySelector("#mail").value;
        let sviPolovi = document.querySelector("#forma").pol;
        let sviJezici = document.querySelector("#forma").jezik;
        let jezici = new Array();
        let selektovaniGrad = document.querySelector("#grad").selectedIndex;

        let regexIme = "/^[A-Z][a-z]{3, 10}$/";
        let regexMail = "/^[A-Za-z0-9.!]{3,15}@[A-Za-z0-9]{2,6}.[a-z]{2,6}$/";

        let nizGresaka = new Array();

        if(ime == "") {
            nizGresaka.push("Problem sa imenom - prazno");
            document.querySelector("pIme").innerHTML = "Ime mora biti popunjeno.";
        }
        else if(!regexIme.test(ime)) {
            nizGresaka.push("Problem sa imenom - nije dobar format");
            document.querySelector("#pIme").innerHTML = "Ime mora poceti velikim slovom, sva ostala mala, u duzini od 3 do 10 karaktera. Primer: Lazar";
        }
        else {
            document.querySelector("pIme").innerHTML = "";
        }

        if(mail == "") {
            nizGresaka.push("Email prazan");
        }
        else if(!regexMail.test(mail)) {
            nizGresaka.push("Email nije dobar format");
        }
        else {
            // ispis korisniku
        }

        if(sviPolovi.count == 0) {
            nizGresaka.push("nije selektovan pol");
        }
        else {
            for(let i=0; i < sviPolovi.count; i++) {
                if(sviPolovi[i].checked) {
                    let pol = sviPolovi[i].value;
                }
            }
        }

        for(let i=0; i< sviJezici.length; i++) {
            if(sviJezici[i].checked) {
                jezik = sviJezici[i].value;
                jezici.push(jezik);
            }
            else {
                jezici.push("Nije selektovao jezik.");
            }
        }
        
        if(selektovaniGrad != 0) {
            let grad = selektovaniGrad.value;
        }
        else {
            nizGresaka.push("Mora se selektovati grad");
        }

        if(nizGresaka.length == 0) {
            $.ajax({
                url: "obrada.php",
                method: "POST",
                dataType: "json",
                data: {
                    imeX: ime,
                    mailX: mail,
                    polX: pol,
                    jeziciX: jezici,
                    gradX: grad
                },
                success: function(data) {
                    console.log(data);
                }
            });
        }
        else {
            console.log(nizGresaka);
        }

    });
});
